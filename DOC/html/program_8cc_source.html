39void actualitzar_ranking()
40{
41 for (int i = 0; i < ranking.size(); ++i) {
42 ranking[i].second = jugadors[ranking[i].first].consultar_punts();
43 }
44
45 // Utilitzarem el bubble sort
46 int n = ranking.size();
47 int i, j;
48 bool canviat = true;
49 for (i = 0; i < n - 1 and canviat; i++) {
50 canviat = false;
51 for (j = 0; j < n - i - 1; j++)
52 if (ranking[j].second < ranking[j + 1].second) {
53 pair<string, int> aux = ranking[j];
54 ranking[j] = ranking[j + 1];
55 ranking[j + 1] = aux;
56 canviat = true;
57 }
58 }
59
60 for (int i = 0; i < ranking.size(); ++i) {
61 jugadors[ranking[i].first].actualitzar_ranking(i + 1);
62 }
63}
